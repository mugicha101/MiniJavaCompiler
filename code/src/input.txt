/**
 * COMP 520
 * int arithmetic
 */
class MainClass {
    public static void main (String [] args) {
        int x = 0;
        while (x < 10) {
            foo(x);
            x = x + 1;
        }
    }

    public static void foo(int x) {
        System.out.println(48 + x);
        System.out.println(32);
        if (x != 5)
            System.out.println(49);
        else
            System.out.println(48);
        System.out.println(10);
    }
}
