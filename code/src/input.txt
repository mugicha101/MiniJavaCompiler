/**
 * COMP 520
 * Field reference and update
 */
class MainClass {
   public static void main (String [] args) {

      FirstClass f = new FirstClass ();
      f.s = new SecondClass ();
      f.s.f = f;

      f.testme ();
   }
}

class FirstClass
{
   int n;
   SecondClass s;

   public void testme ()
   {
      int tstvar = 48;
      System.out.println(tstvar);
      n = 49;
      System.out.println(n);
      s = s;
      System.out.println(48);
      s.f = s.f;
      System.out.println(49);
      s.f.n = s.f.n;
      System.out.println(50);
      tstvar = s.f.n;
      System.out.println(tstvar);
	  System.out.println(tstvar);
   }

}

class SecondClass
{
   int n;
   FirstClass f;
}
